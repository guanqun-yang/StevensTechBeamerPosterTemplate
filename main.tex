  \begingroup
  \setbeamercolor{block title}{fg=#1, bg=#2}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% POSTER BODY - figure-centric 3-column layout
%
%   left column  (0.235w): introduction / method narrative
%   center col   (0.510w): hero diagram up top, then two text sub-columns
%                          for algorithm / setup / formulation
%   right column (0.235w): results (with takeaway boxes) + take-home
%
% Reading order flows column-by-column, sub-column-by-sub-column:
%   left -> center top figure -> center sub-left -> center sub-right -> right
%
% Why this layout beats a flat multicols flow:
%   - the diagram lands in the visual center of the poster, where the
%     reader's eye naturally goes first
%   - results stay together in a dedicated narrow column, so each result
%     block + takeaway reads as a self-contained unit
%   - narrow side columns force shorter lines, which is easier to scan
%     at standing distance than 4 equal-width columns
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\posterbody}{%
  \begin{columns}[T]
    \begin{column}{0.235\textwidth}
      \input{sections/001-motivation.tex}
      \input{sections/002-design.tex}
      \input{sections/003-example.tex}
    \end{column}%
    \hfill
    \begin{column}{0.510\textwidth}
      \begin{center}
        \scalebox{1}[0.5]{\includegraphics[width=\linewidth]{example-image}}
      \end{center}
      \vspace{0.6em}
      \setlength{\columnsep}{2cm}%
      \begin{multicols}{2}
        \input{sections/004-algorithm.tex}
        \columnbreak
        \input{sections/005-setup.tex}
      \end{multicols}
    \end{column}%
    \hfill
    \begin{column}{0.235\textwidth}
      \input{sections/006-results.tex}
      \input{sections/007-takehome.tex}
    \end{column}
  \end{columns}%
}
