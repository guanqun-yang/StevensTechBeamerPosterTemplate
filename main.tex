% colorblock: \begin{colorblock}[text color]{background color}{title}
\newenvironment{colorblock}[3][black]{%
  \begingroup
  \setbeamercolor{block title}{fg=#1, bg=#2}
  \setbeamercolor{block body}{fg=#1, bg=#2}
  \begin{block}{#3}
}{%
  \end{block}
  \endgroup
}

% contribblock: colored header (white text), light gray body (black text)
\definecolor{contribbg}{RGB}{245,245,245}
\newenvironment{contribblock}[1]{%
  \begingroup
  \setbeamercolor{block title}{fg=white, bg=stevensred}
  \setbeamercolor{block body}{fg=black, bg=contribbg}
  \begin{block}{#1}
}{%
  \end{block}
  \endgroup
}

\title{Your Poster Title Here}

% Multi-institution authors: use \inst{N} after each name, with N referring to
% the institute index below. An author with multiple affiliations writes
% \inst{1,3}. Authors are comma-separated on a single line.
\author{%
  First Author\inst{1},\enspace
  Second Author\inst{2},\enspace
  Third Author\inst{1,3}%
}

% Each institute is comma-separated and prefixed with a matching \inst{N}.
% (Using literal commas instead of \and avoids beamerposter's title-template
% inserting line breaks.)
\institute{%
  \inst{1}Stevens Institute of Technology,\enspace
  \inst{2}Partner University,\enspace
  \inst{3}Third Institution%
}

% POSTER HEADER (full-width banner)

% Right-corner content: 2 placeholder images side-by-side, 1 centered, or empty.
% (To use real QR codes instead, \usepackage{qrcode} and replace each
%  \includegraphics{example-image-*} with \qrcode[height=...]{\projecturl}.)
\newcommand{\posterrightcorner}{%
  \ifshowprojectqr
    \ifshowcodeqr
      % --- both: side-by-side, each fills its half-slot ---
      \begin{minipage}[c]{0.48\linewidth}
        \centering
        \includegraphics[width=\linewidth]{example-image-a}\\[0.3em]
        {\footnotesize Project page}
      \end{minipage}\hfill
      \begin{minipage}[c]{0.48\linewidth}
        \centering
        \includegraphics[width=\linewidth]{example-image-b}\\[0.3em]
        {\footnotesize Code}
      \end{minipage}%
    \else
      % --- only project: centered, fills most of the corner ---
      \centering
      \includegraphics[width=0.85\linewidth]{example-image-a}\\[0.3em]
      {\footnotesize Project page}%
    \fi
  \else
    \ifshowcodeqr
      % --- only code: centered, fills most of the corner ---
      \centering
      \includegraphics[width=0.85\linewidth]{example-image-b}\\[0.3em]
      {\footnotesize Code}%
    \fi
    % --- neither: corner stays empty, preserves title centering ---
  \fi
}

\newcommand{\posterheader}{%
  \begin{minipage}[c]{0.22\linewidth}
    \includegraphics[width=\linewidth]{stevens-long-logo}
  \end{minipage}\hfill
  \begin{minipage}[c]{0.54\linewidth}
    \centering
    {\veryHuge\bfseries\color{stevensred}\inserttitle\par}
    \vspace{0.3em}
    {\Large\insertauthor\par}
    \vspace{0.2em}
    {\normalsize\insertinstitute\par}
  \end{minipage}\hfill
  \begin{minipage}[c]{0.22\linewidth}
    \posterrightcorner
  \end{minipage}%
  \par\vspace{0.4em}
  {\color{stevensred}\rule{\linewidth}{2pt}}
}

% Footline template: red rule + contact strip, always anchored at page bottom
\setbeamertemplate{footline}{%
  \leavevmode%
  \begin{minipage}{\paperwidth}%
    \hspace*{2em}%
    \begin{minipage}{\dimexpr\paperwidth-4em\relax}%
      {\color{stevensred}\rule{\linewidth}{1pt}}\par
      \vspace{0.3em}
      \centering\small
      Stevens Institute of Technology \quad|\quad
      \hrefcol{\projecturl}{\projectlabel} \quad|\quad
      \hrefcol{\codeurl}{\codelabel} \quad|\quad
      \hrefcol{https://www.stevens.edu/}{stevens.edu} \quad|\quad
      \insertdate
    \end{minipage}%
  \end{minipage}%
  \vspace{1em}%
}

% Number of body columns: 4 for landscape, 3 for portrait.
\ifposterlandscape
  \newcommand{\posternumcols}{4}
\else
  \newcommand{\posternumcols}{3}
\fi

% All sections are concatenated into a single multicols stream and flowed
% across columns automatically. Each beamer block is wrapped in a minipage
% (see preamble) so multicol treats it as one atomic unit -- blocks move
% between columns whole, never sliced mid-content. No manual \filbreak
% needed in the section files.
\newcommand{\posterbody}{%
  \setlength{\columnsep}{3cm}%
  \setlength{\columnseprule}{0pt}%
  \begin{multicols}{\posternumcols}
    \input{sections/001-introduction.tex}
    \input{sections/002-methods.tex}
    \input{sections/003-results.tex}
    \input{sections/004-conclusion.tex}
  \end{multicols}%
}
